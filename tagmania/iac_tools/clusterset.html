 10    def __init__(self, cluster_names, profile=None):
 11        """
 12        Initialize ClusterSet.
 13
 14        Args:
 15            - cluster_names - The name of a cluster, or a list of clusters
 16            - profile - The AWS profile to use (optional)
 17        Returns:
 18            none
 19        """
 20        # Collections operate lazily. Turning a collection into a list can cause
 21        # performance issues if the collection is very large. Although this is
 22        # unlikely in our environment, we protect against this by setting an
 23        # upper bound on the number of items from the collection that can be
 24        # placed in the list. If the number of items in the collection exceed
 25        # this upper bound, then those items will not be processed. If this
 26        # happens, then it can be addressed by increasing this value.
 27        self._MAX_ITEMS = 150
 28
 29        # Used to ensure we don't clobber anything we don't make
 30        self.AUTOMATION_KEY = 'SNAPSHOT_MANAGER'
 31
 32        # cluster_names can be a string 'aws-dev5' or a list ['aws-dev1', 'aws-dev2', etc...]
 33        self.cluster_names = cluster_names
 34
 35        if not isinstance(cluster_names, list):
 36            cluster_names = [self.cluster_names]
 37
 38        # Use this as a starting point to build filters that only return
 39        # resources associated with this cluster. Use the 'get_cluster_filter'
 40        # method to get a copy of it rather than doing direct assignment.
 41        self._cluster_filter = [
 42            {'Name': 'tag:Cluster', 'Values': cluster_names},
 43        ]
 44
 45        # Set up logging
 46        self._logger = logging.getLogger('tagmania')
 47        self._logger.setLevel(logging.INFO)
 48        self._logger.info("Logging initialized.")
 49
 50        # Create a boto3 session using the specified profile if provided.
 51        if profile:
 52            aws_session = boto3.Session(profile_name=profile)
 53            self._logger.info(f"Using AWS profile: {profile}")
 54        else:
 55            aws_session = boto3.Session()
 56
 57        # Create EC2 resource and client from the session.
 58        self._ec2 = aws_session.resource('ec2')
 59        self._ec2_client = aws_session.client('ec2')

371        self._logger.debug("method_call: get_restored_volumes")

391    def attach_volumes(self, label):
392        """
393        Attach volumes to associated instances.
394
395        Args:
396            - label: label of volume to attach
397        Returns:
398            none
399        """
400        self._logger.debug("method_call: attach_volumes")
401        instances = self.get_instances()
402        # We should be able to work with get_volumes, but this just protects
403        # against the case when users are manually creating volumes. It filters
404        # out any volumes that were not created by the snapshot manager.
405        volumes = self.get_restored_volumes(label)
406        # This is for debugging purposes. Sometimes the algorithm below doesn't
407        # find all volumes.
408        # volume_list = list(volumes)
409        # print(f"Found {len(volume_list)} volumes to attach.")
410
411        # For each instance, find all associated volumes and attach them. The
412        # association is performed by matching the volume 'Instance' tag to
413        # the instance 'Name' tag.
414        volume_ids = []
415        for i in instances:
416            instance_name = TagSet(i.tags).get('Name')
417            for volume in volumes:
418                ts = TagSet(volume.tags)
419                instance = ts.get('Instance')
420                device = ts.get('Device')
421                if instance == instance_name:
422                    # Attach volume
423                    shortname = instance_name.split('.')[0]
424                    print(f"Attaching {device} ({volume.id}) to {shortname} ({i.id})")
425                    volume.attach_to_instance(Device=device, InstanceId=i.id)
426                    volume_ids.append(volume.id)
427        if len(volume_ids) == 0:
428            # This is probably an error. The expectation is that we have a set
429            # of newly created volumes from snapshots.
430            print("Error: No volumes to attach.")
431        else:
432            # Wait for the volumes to be attached
433            print(f"Waiting for {len(volume_ids)} volumes to be attached...")
434            self.wait_for_volumes(volume_ids, 'volume_in_use')

436    def create_volumes(self, label):
437        """
438        Create new volumes from managed snapshots.
439
440        Args:
441            - label: label of snapshots to restore
442        Returns:
443            none
444        """
445        self._logger.debug("method_call: create_managed_volumes")
446        snapshots = self.get_snapshots(label)
447        # Check if snapshot list is empty (e.g. due to an invalid label)
448        if len(snapshots) == 0:
449            print(f"Error: No snapshots found with label '{label}'.")
450        # Create volumes
451        volume_ids = []
452        for snapshot in snapshots:
453            # Determine snapshot's associated instance and device. This is
454            # needed later on so that we know where to attach it.
455            ts = TagSet(snapshot.tags)
456            device = ts.get('Device')
457            instance = ts.get('Instance')
458            if not device:
459                raise Exception(f"Error: create_volume: Can't find device tag for snapshot {snapshot.id}.")
460            if not instance:
461                raise Exception(f"Error: create_volume: Can't find instance tag for snapshot {snapshot.id}.")
462            # Determine availability zone from one of the cluster instances
463            avail_zone = self.get_instances()[0].placement['AvailabilityZone']
464            # Make tags
465            ts = TagSet()
466            ts.add('Cluster', self.cluster_names)
467            ts.add('Device', device)
468            ts.add('Instance', instance)
469            ts.add('Label', label)
470            ts.add('Name', f"{instance} - {device}")
471            ts.add('automation_key', self.AUTOMATION_KEY)
472            tags = ts.to_list()
473            # Create volume
474            print(f"Creating volume from snapshot {snapshot.id}")
475            volume = self._ec2.create_volume(
476                SnapshotId=snapshot.id,
477                AvailabilityZone=avail_zone,
478                TagSpecifications=[{'ResourceType': 'volume', 'Tags': tags}]
479            )
480            volume_ids.append(volume.id)
481        # Wait for the volumes to be created
482        if len(volume_ids) > 0:
483            print(f"Waiting for {len(volume_ids)} volumes to be created...")
484            self.wait_for_volumes(volume_ids, 'volume_available')
485            # When attaching newly created volumes later, not all volumes are
486            # being attached for some reason. Perhaps the waiter returns before
487            # the tags have actually been applied. Try giving ten more seconds
488            # for the tags to be applied.
489            time.sleep(10)

638    def create_snapshots(self, label):
639        """
640        Create snapshots of volumes.
641
642        Args:
643            - label: label to apply to each snapshot
644        Returns:
645            none
646        """
647        self._logger.debug("method_call: create_snapshots")
648        # Check if any snapshots with the same label already exists. If so,
649        # delete them. Only one set of snapshots with a given label may
650        # exist at a time.
651        old_snapshots = self.get_snapshots(label)
652        if len(old_snapshots) > 0:
653            self.delete_snapshots(label)
654        snapshot_ids = []
655        # Get list of instances that need snapshots taken
656        instances = self.get_instances()
657        for i in instances:

437    def create_volumes(self, label):
438        """
439        Create new volumes from managed snapshots.
440
441        Args:
442            - label: label of snapshots to restore
443        Returns:
444            none
445        """
446        self._logger.debug("method_call: create_managed_volumes")
447        snapshots = self.get_snapshots(label)
448        # Check if snapshot list is empty (e.g. due to an invalid label)
449        if len(snapshots) == 0:
450            print(f"Error: No snapshots found with label '{label}'.")
451        # Create volumes
452        volume_ids = []
453        for snapshot in snapshots:
454            # Determine snapshot's associated instance and device. This is
455            # needed later on so that we know where to attach it.
456            ts = TagSet(snapshot.tags)
457            device = ts.get('Device')
458            instance = ts.get('Instance')
459            if not device:
460                raise Exception(f"Error: create_volume: Can't find device tag for snapshot {snapshot.id}.")
461            if not instance:
462                raise Exception(f"Error: create_volume: Can't find instance tag for snapshot {snapshot.id}.")
463            # Determine availability zone from one of the cluster instances
464            avail_zone = self.get_instances()[0].placement['AvailabilityZone']
465            # Make tags
466            ts = TagSet()
467            ts.add('Cluster', self.cluster_names)
468            ts.add('Device', device)
469            ts.add('Instance', instance)
470            ts.add('Label', label)
471            ts.add('Name', f"{instance} - {device}")
472            ts.add('automation_key', self.AUTOMATION_KEY)
473            tags = ts.to_list()
474            # Create volume
475            print(f"Creating volume from snapshot {snapshot.id}")
476            volume = self._ec2.create_volume(
477                SnapshotId=snapshot.id,
478                AvailabilityZone=avail_zone,
479                TagSpecifications=[{'ResourceType': 'volume', 'Tags': tags}]
480            )
481            volume_ids.append(volume.id)
482        # Wait for the volumes to be created
483        if len(volume_ids) > 0:
484            print(f"Waiting for {len(volume_ids)} volumes to be created...")
485            self.wait_for_volumes(volume_ids, 'volume_available')
486            # When attaching newly created volumes later, not all volumes are
487            # being attached for some reason. Perhaps the waiter returns before
488            # the tags have actually been applied. Try giving ten more seconds
489            # for the tags to be applied.
490            time.sleep(10)
